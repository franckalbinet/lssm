Training

def set_grad(m, b):
    if isinstance(m, (nn.Linear,nn.BatchNorm2d)): return
    if hasattr(m, 'weight'):
        for p in m.parameters(): p.requires_grad_(b)
def get_n_params(model, trainable=True):
    return sum(p.numel() for p in model.parameters() if p.requires_grad == trainable)
